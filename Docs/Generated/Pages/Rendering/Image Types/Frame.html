class Frame extends ImageType
Represents an offscreen drawing surface that can be rendered as an image. const frame = new Frame(100, 200); // add shapes to the frame frame.renderer.stroke(new Color("blue"), 2).rect(10, 10, 20, 20); frame.renderer.draw(new Color("red")).circle(30, 100, 50); // render the frame to the screen renderer.image(frame).default(0, 0);
